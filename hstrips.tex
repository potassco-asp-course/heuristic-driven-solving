\begin{frame}[fragile,shrink=1]{Simple STRIPS planning}
\begin{semiverbatim}
time(1..k).

holds(P,0) :- init(P).

\{ occ(A,T) : action(A) \} = 1 :- time(T).
:- occ(A,T), pre(A,F), not holds(F,T-1).

holds(F,T) :- occ(A,T), add(A,F).
holds(F,T) :- holds(F,T-1), time(T), not occ(A,T) : del(A,F).

:- query(F), not holds(F,k).

\only<1>{

}\only<2-4>{\#heuristic occurs(A,T) : action(A), time(T). [\only<2>{2, factor}\only<3>{1, level}\only<4>{T, factor}]

}\only<5>{\#heuristic holds(F,T-1) :     holds(F,T). [t-T+1, true]
\#heuristic holds(F,T-1) : not holds(F,T)  [t-T+1, false]
                          fluent(F), time(T).}

\end{semiverbatim}
\end{frame}
